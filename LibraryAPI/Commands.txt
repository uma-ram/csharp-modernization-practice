Docker Commands for TodoApi
- docker build -t todoapi:latest .
- docker stop todoapi_container || true
- docker rm todoapi_container || true
- docker run -d -p 5000:8080 --name todoapi_container -e ASPNETCORE_ENVIRONMENT=Development todoapi:latest

Docker Commands for Library-API

- docker build -t library-api:v1 .
- docker stop library-container || true
- docker rm library-container || true
- docker run -d -p 5001:8080 --name library-container -e ASPNETCORE_ENVIRONMENT=Development library-api:v1
